#define DBNAME @"user_cache_data.db"
static FMDatabaseQueue *_queue;

@implementation UserCacheInfo

@implementation UserCacheManager

+(void)initialize{
    NSString *fileName = [[NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES)lastObject] stringByAppendingPathComponent:DBNAME];
    
    _queue = [FMDatabaseQueue databaseQueueWithPath:fileName];
    [_queue inDatabase:^(FMDatabase *db) {
        // userid 环信ID，username 用户昵称，userimage 用户头像完整路径, staffid   员工ID
        [db executeUpdate:@"create table if not exists userinfo (userid text, username text, userimage text, staffid integer)"];
    }];
}

/**
 *  执行一个更新语句
 *
 *  @param sql 更新语句的sql
 *
 *  @return 更新语句的执行结果
 */
+(BOOL)executeUpdate:(NSString *)sql{
    
    __block BOOL updateRes = NO;
    
    [_queue inDatabase:^(FMDatabase *db) {
        
        updateRes = [db executeUpdate:sql];
    }];
    
    return updateRes;
}


/**
 *  执行一个查询语句
 *
 *  @param sql              查询语句sql
 *  @param queryResBlock    查询语句的执行结果
 */
+(void)executeQuery:(NSString *)sql queryResBlock:(void(^)(FMResultSet *set))queryResBlock{
    
    [_queue inDatabase:^(FMDatabase *db) {
        
        FMResultSet *set = [db executeQuery:sql];
        
        if(queryResBlock != nil) queryResBlock(set);
        
    }];
}

/**
 *  用户是否存在
 *
 *  @param userId 用户环信ID
 *
 *  @return 是否存在
 */
+(BOOL)isExisted:(NSString *)userId{
    
    NSString *alias=@"count";
    
    NSString *sql = [NSString stringWithFormat:@"SELECT COUNT(*) AS %@ FROM userinfo where userid = '%@'", alias, userId];
    
    __block NSUInteger count=0;
    
    [self executeQuery:sql queryResBlock:^(FMResultSet *set) {
        
        while ([set next]) {
            
            count = [[set stringForColumn:alias] integerValue];
        }
    }];
    
    return count > 0;
}



/**
 清除数据
 @return 是否成功
 */
+(BOOL)clearData{
    
    BOOL isSucc = [self executeUpdate:@"DELETE FROM userinfo"];
    [self executeUpdate:@"DELETE FROM sqlite_sequence WHERE name='userinfo';"];
    return isSucc;
}


/**
 保存（新增或更新）用户信息
 @param userId 用户环信ID
 @param avatarUrl 头像Url
 @param nickName 昵称
 */
+(void)save:(NSString*)userId
    staffId:(NSNumber*)staffId
  avatarUrl:(NSString*)avatarUrl
   nickName:(NSString*)nickName{
    
    if(!userId) return;
    
    NSString *sql = @"";
    
    BOOL isExisted = [self isExisted:userId];
    if (isExisted) {
        sql = [NSString stringWithFormat:@"update userinfo set username='%@', userimage='%@', staffid='%ld' where userid='%@'", nickName,avatarUrl,staffId.integerValue,userId];
    }else{
        sql = [NSString stringWithFormat:@"INSERT INTO userinfo (userid, username, userimage, staffid) VALUES ('%@', '%@', '%@', '%ld')", userId,nickName,avatarUrl,staffId.integerValue];
    }
    
    [self executeUpdate:sql];
    
#if DEBUG
//    [self queryAll];
#endif
}

/*
 *根据环信ID获取用户信息
 *userId 用户的环信ID
 */
+(UserCacheInfo*)getFromCache:(NSString *)userid{
    
    __block UserCacheInfo *userInfo = nil;
    
    NSString *sql = [NSString stringWithFormat:@"SELECT userid, username, userimage,staffid FROM userinfo where userid = '%@'",userid];
    [self executeQuery:sql queryResBlock:^(FMResultSet *rs) {
        if ([rs next]) {
            
            userInfo = [[UserCacheInfo alloc] init];
            userInfo.userId = [rs stringForColumn:@"userid"];
            userInfo.nickName = [rs stringForColumn:@"username"];
            userInfo.avatarUrl = [rs stringForColumn:@"userimage"];
            userInfo.staffId = [NSNumber numberWithInteger:[[rs stringForColumn:@"staffid"] integerValue]];
        }
        [rs close];
    }];
    
    return userInfo;
}

/**
 根据环信ID获取昵称
 @param userId 用户的环信ID
 @return 昵称
 */
+(NSString*)getNickName:(NSString*)userId{
    UserCacheInfo *user = [UserCacheManager getUserInfo:userId];
    if(user == nil || [user  isEqual: @""]) return userId;// 没有昵称就返回用户环信ID
    
    return user.nickName;
}

/**
 根据环信ID获取员工ID

 */
+(NSNumber*)getStaffId:(NSString *)userId{
    UserCacheInfo *user = [UserCacheManager getUserInfo:userId];
    if(user == nil || [user  isEqual: @""]) return [NSNumber numberWithInteger:0];// 没有昵称就返回用户环信ID
    
    return user.staffId;
}

// 设置头像昵称控件
+(void)setUserView:(NSString*)userId
               nickLabel:(UILabel*)nameLabel
               imageView:(UIImageView*)imageView
          imageNameLabel:(UILabel*)imageNameLabel{
    [UserCacheManager getUserInfo:userId completed:^(UserCacheInfo *userInfo) {
        if (userInfo) {
            if (userInfo.avatarUrl && ![userInfo.avatarUrl isEqualToString:@""]) {
                [imageView sd_setImageWithURL:[NSURL URLWithString:userInfo.avatarUrl] placeholderImage:[UIImage imageNamed:@"icon_personalavatar"]];
                [imageNameLabel setHidden:YES];
            } else {
                imageView.image = [UIImage imageNamed:@"icon_personalavatar"];
                [imageNameLabel setHidden:YES];
//                imageNameLabel.text = [userInfo.nickName substringToIndex:1];
            }
            [nameLabel setText:userInfo.nickName];
            
        } else {
            [nameLabel setText:userId];
            imageView.image = [UIImage imageNamed:@"icon_personalavatar"];
            [imageNameLabel setHidden:YES];
//            imageNameLabel.text = [userId substringToIndex:1];
        }
    }];
}
